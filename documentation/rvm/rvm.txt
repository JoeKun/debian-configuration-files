# Install
cd ~root
ln -s /debian-configuration-files/root/.rvmrc
curl -L https://get.rvm.io | bash
rm -f /etc/profile.d/rvm.sh /etc/rvmrc
logout
rvm install 2.0.0 --patch /debian-configuration-files/documentation/rvm/ruby-2.0.0-p0_missing_declarations_GNU_kFreeBSD.diff

# System gemsets
rvm gemset create integrity_check
rvm gemset use integrity_check
gem install activerecord sqlite3

# Uninstall
rm -R -f /etc/profile.d/rvm.sh /etc/rvmrc /usr/local/rvm /usr/local/share/man/man1/rvm* ~root/.rvm ~root/.gem ~root/.rvmrc
